## 💻 技术实现

### 1. 套餐配置

```javascript
const AI_CONFIG = {
    TIER_LIMITS: {
        'free': 3,      // 免费体验 - 3 个名字
        'premium': 10,  // 深度体验版 - 10 个名字
        'deluxe': 30    // 定制臻选版 - 30 个名字
    }
};
```

### 2. 套餐存储

```javascript
// 从 localStorage 读取当前套餐
let currentUserTier = localStorage.getItem('userTier') || 'free';

// 用户选择套餐后保存
function selectPricing(tier) {
    currentUserTier = tier;
    localStorage.setItem('userTier', tier);
}
```

### 3. 动态 Prompt

```javascript
function buildNamingPrompt(formData) {
    const nameCount = getNameLimitByTier(); // 根据套餐决定数量
    
    return `请为以下宝宝起${nameCount}个名字：
    
【宝宝信息】
- 姓氏：${formData.surname}
- 性别：...
- 套餐类型：${currentUserTier === 'free' ? '免费体验' : ...}

【起名要求】
${currentUserTier !== 'free' ? '- 每个字的独立解析' : ''}
...
`;
}
```

### 4. 套餐展示

```javascript
function updateTierDisplay() {
    // 在表单上方显示当前套餐徽章
    const tierBadge = document.createElement('div');
    tierBadge.textContent = `当前套餐：${tierNames[currentUserTier]}（${getNameLimitByTier()}个名字）`;
    formSection.insertBefore(tierBadge, formSection.firstChild);
}
```
