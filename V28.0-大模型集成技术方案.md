## 代码实现

### 1. API 调用函数

```javascript
/**
 * 调用大模型生成名字
 * @param {Object} formData - 用户表单数据
 * @returns {Promise<Array>} 名字数组
 */
async function generateNamesByAI(formData) {
    const prompt = buildNamingPrompt(formData);
    
    const API_KEY = process.env.DASHSCOPE_API_KEY; // 从环境变量读取
    const API_ENDPOINT = 'https://dashscope.aliyuncs.com/api/v1/services/aigc/text-generation/generation';
    
    const requestBody = {
        model: 'qwen-max',
        input: {
            messages: [
                {
                    role: 'system',
                    content: '你是一位专业起名专家，精通《诗经》《楚辞》《周易》等 20+ 中华经典。请根据用户提供的信息生成 10 个有文化底蕴的好名字，严格返回 JSON 格式。'
                },
                {
                    role: 'user',
                    content: prompt
                }
            ]
        },
        parameters: {
            result_format: 'json',
            temperature: 0.7,
            max_tokens: 2000
        }
    };
    
    try {
        const response = await fetch(API_ENDPOINT, {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${API_KEY}`,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(requestBody)
        });
        
        if (!response.ok) {
            const error = await response.json();
            throw new Error(`API 调用失败：${error.message || response.statusText}`);
        }
        
        const result = await response.json();
        const aiText = result.output.text;
        
        // 解析 AI 返回的 JSON
        const names = parseAIResponse(aiText);
        
        // 补充八字信息（前端计算）
        const baziInfo = calculateBazi(formData.birthDate, formData.birthTime);
        names.forEach(name => {
            name.bazi = baziInfo;
        });
        
        return names;
        
    } catch (error) {
        console.error('大模型调用失败:', error);
        throw error;
    }
}
```

### 2. 结果解析函数

```javascript
/**
 * 解析大模型返回的 JSON
 * @param {string} text - AI 返回的文本
 * @returns {Array} 名字数组
 */
function parseAIResponse(text) {
    try {
        // 尝试提取 JSON（AI 可能返回多余文字）
        let jsonText = text.trim();
        
        // 如果文本不是以 { 开头，尝试提取 JSON 块
        if (!jsonText.startsWith('{')) {
            const jsonMatch = text.match(/\{[\s\S]*\}/);
            if (jsonMatch) {
                jsonText = jsonMatch[0];
            } else {
                throw new Error('未找到 JSON 格式数据');
            }
        }
        
        const data = JSON.parse(jsonText);
        
        // 验证数据结构
        if (!data.names || !Array.isArray(data.names)) {
            throw new Error('数据格式错误：缺少 names 数组');
        }
        
        // 验证并补充字段
        const validatedNames = data.names.map((name, index) => ({
            name: name.name || `名字${index + 1}`,
            score: name.score || (90 - index), // 默认递减评分
            source: name.source || '典籍出处待补充',
            originalText: name.originalText || '原文待补充',
            meaning: name.meaning || '寓意待解读',
            pinyin: name.pinyin || 'pīn yīn',
            tone: name.tone || '声调待补充',
            pingze: name.pingze || '平仄待补充',
            yinlv: name.yinlv || '音律待补充',
            wuxing: name.wuxing || '五行待补充'
        }));
        
        return validatedNames;
        
    } catch (error) {
        console.error('JSON 解析失败:', error);
        console.error('原始文本:', text);
        
        // 降级到备用方案
        return getFallbackNames();
    }
}
```

### 3. 八字计算函数（简化版）

```javascript
/**
 * 计算生辰八字（简化版，真实场景需要精确节气计算）
 * @param {string} birthDate - 出生日期 YYYY-MM-DD
 * @param {string} birthTime - 出生时间 HH:mm
 * @returns {Object} 八字信息
 */
function calculateBazi(birthDate, birthTime) {
    const date = new Date(birthDate);
    const year = date.getFullYear();
    
    // 简化版：按农历年份推算天干地支
    const heavenlyStems = ['甲', '乙', '丙', '丁', '戊', '己', '庚', '辛', '壬', '癸'];
    const earthlyBranches = ['子', '丑', '寅', '卯', '辰', '巳', '午', '未', '申', '酉', '戌', '亥'];
    
    const yearIndex = (year - 4) % 60;
    const stemIndex = yearIndex % 10;
    const branchIndex = yearIndex % 12;
    
    const yearPillar = heavenlyStems[stemIndex] + earthlyBranches[branchIndex];
    
    // 简化处理：月日时柱需要精确节气计算，这里用占位
    return {
        year: yearPillar,
        month: '待精确计算',
        day: '待精确计算',
        hour: '待精确计算',
        analysis: '八字分析需要精确节气数据，建议接入专业八字 API'
    };
}
```

## 限流策略

### 前端限流（LocalStorage）

```javascript
/**
 * 检查用户是否超出使用限制
 * @returns {boolean} 是否可以调用
 */
function canUseAI() {
    const usage = JSON.parse(localStorage.getItem('aiUsage') || '{"count":0,"date":"2026-04-18"}');
    const today = new Date().toISOString().split('T')[0];
    
    // 如果是新的一天，重置计数
    if (usage.date !== today) {
        usage.count = 0;
        usage.date = today;
    }
    
    // 检查是否超出限制（免费用户 3 次/天）
    const dailyLimit = 3;
    return usage.count < dailyLimit;
}

/**
 * 记录使用次数
 */
function recordUsage() {
    const usage = JSON.parse(localStorage.getItem('aiUsage') || '{"count":0,"date":"2026-04-18"}');
    const today = new Date().toISOString().split('T')[0];
    
    if (usage.date !== today) {
        usage.count = 0;
        usage.date = today;
    }
    
    usage.count++;
    localStorage.setItem('aiUsage', JSON.stringify(usage));
}

/**
 * 获取剩余使用次数
 * @returns {number} 剩余次数
 */
function getRemainingUses() {
    const usage = JSON.parse(localStorage.getItem('aiUsage') || '{"count":0,"date":"2026-04-18"}');
    const today = new Date().toISOString().split('T')[0];
    
    if (usage.date !== today) {
        return 3;
    }
    
    return Math.max(0, 3 - usage.count);
}
```

/**

function showUserFriendlyError(message) {
    const toast = document.createElement('div');
    toast.className = 'error-toast';
    toast.style.cssText = `
        position: fixed;
        top: 20px;
        left: 50%;
        transform: translateX(-50%);
        background: rgba(196, 30, 58, 0.95);
        color: white;

    document.body.appendChild(toast);
    
    setTimeout(() => {
        toast.style.transition = 'opacity 0.3s';
        toast.style.opacity = '0';
        setTimeout(() => toast.remove(), 300);
    }, 3000);
}

/**
 * 处理 DashScope API 错误

function handleAPIError(error) {
    // DashScope 常见错误码
    const errorMessages = {
        'InvalidApiKey': 'API 密钥无效，请联系管理员',
        'QuotaExhausted': '今日调用次数已用完，请明日再来',
        'RateLimitExceeded': '请求过于频繁，请稍后再试',
        'InsufficientBalance': '账户余额不足，请充值',
        'InvalidParameter': '请求参数错误',
        'ServiceUnavailable': '服务暂时不可用，请稍后重试'
    };
    
    const errorCode = error.code || 'UnknownError';
    const message = errorMessages[errorCode] || `服务异常：${errorCode}`;
    
    showUserFriendlyError(message);
    
    // 记录错误日志
    console.error('API 错误:', {
        code: errorCode,
        message: error.message,
        requestId: error.request_id
    });
}
